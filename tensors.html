Converting between Tensors and arrays

System.Array and F# arrays

open Furnace

// Tensor
let t1 = FurnaceImage.tensor [ 0.0 .. 0.2 .. 1.0 ]

// System.Array
let a1 = t1.toArray()

// []<float32>
let a1b = t1.toArray() :?> float32[]

// Tensor
let t2 = FurnaceImage.randn([3;3;3])

// [,,]<float32>
let a2 = t2.toArray() :?> float32[,,]
